export const FilterableComponentsList = () => {
  const [filter, setFilter] = useState('')
  const filtered = !filter.trim()
    ? components
    : components.filter((c) => {
        const lower = filter.toLowerCase().trim()
        return (
          c.name.toLowerCase().includes(lower) ||
          (c.description && c.description.toLowerCase().includes(lower))
        )
      })
  return (
    <>
      <div className="sb-unstyled mb-xl max-w-sz-320">
        <Input
          placeholder="Button, Input, etc."
          value={filter}
          onValueChange={setFilter}
          aria-label="Filter components by name"
        />
      </div>
      <div className="sb-unstyled my-xl gap-lg flex-wrap flex">
        {filtered.length === 0 ? (
          <p className="text-body-1 text-neutral-outline">
            No components match &quot;{filter}&quot;. Try a different search.
          </p>
        ) : (
          filtered.map((component) => (
            <div key={component.name} className="flex flex-col gap-sm w-sz-184 shrink-0 min-h-[100px]">
              <a href={component.href} className="flex gap-md flex-col group">
                <TextLink underline={false} intent="basic" className="text-headline-2">
                  {component.name}
                </TextLink>

                <div className="group-hover:scale-110 group-hover:shadow-sm transition-transform duration-150 w-[184px] h-[100px] bg-neutral-container rounded-lg flex items-center justify-center">
                  <img
                    src={component.thumbnails?.light}
                    alt={`${component.name} component thumbnail`}
                    className="w-full h-full object-contain"
                    loading="lazy"
                    width="184"
                    height="100"
                  />
                </div>
              </a>
            </div>
          ))
        )}
      </div>
    </>
  )
}
