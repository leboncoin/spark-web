<div className="sb-unstyled my-xl gap-lg grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3">
  {components.map((component) => (
    <LinkBox asChild>
      <article className="flex gap-md flex-col p-lg rounded-lg border-sm border-outline">
      
        <div className="flex gap-md items-start">
            <div className="bg-neutral-container size-sz-96" />

            <LinkBox.Link href={component.href} asChild className="text-headline-2">
                <TextLink underline={false} intent="main">
                    {component.name}
                </TextLink>
            </LinkBox.Link>  
        </div>

        {component.related && (
          <div className="flex gap-md text-body-2">
            <span>Related:</span>
            {component.related.map((related) => {
              const relatedComponent = components.find((c) => c.name === related)
              return (
                <LinkBox.Raised>
                  <TextLink intent="info" href={relatedComponent.href}>
                    {relatedComponent.name}
                  </TextLink>
                </LinkBox.Raised>
              )
            })}
          </div>
        )}
        

        {component.deps && (
          <div className="flex gap-md absolute top-0 -translate-y-1/2 left-lg">
            {component.deps.map((dep) => {
              
              return (
                <LinkBox.Raised>
                  
                    <Tag intent="neutral" design="outlined" asChild className='focus-visible:u-outline bg-surface'>
                      <a href={dep.href} target="_blank">{dep.name}</a>
                    </Tag>
                  
                </LinkBox.Raised>
              )
            })}
          </div>
        )}
      </article>
    </LinkBox>
  ))}
</div>
