Spark offers {components.length} components and the list will grow, here is the list:

<div className="sb-unstyled my-xl gap-lg flex-wrap flex">
  {components.map((component) => (
    <div className='flex flex-col gap-sm max-w-sz-240'>
      <LinkBox asChild>
        <article className="flex gap-md flex-col">
          <LinkBox.Link href={component.href} asChild className="text-headline-2">
              <TextLink underline={false} intent="basic">
                  {component.name}
              </TextLink>
          </LinkBox.Link>  
          <img src={component.thumbnails?.light} className="w-full h-auto object-contain" />
        </article>
      </LinkBox>
      <div>
        {component.related && (
          <div className="flex gap-md text-caption">
            <span>Related:</span>
            {component.related.map((related) => {
              const relatedComponent = components.find((c) => c.name === related)
              return (
                <LinkBox.Raised>
                  <TextLink intent="info" href={relatedComponent.href}>
                    {relatedComponent.name}
                  </TextLink>
                </LinkBox.Raised>
              )
            })}
          </div>
        )}
      </div>
    </div>
  ))}
</div>
