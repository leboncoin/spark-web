export const FilterableComponentsList = ({ filterType = null } = props) => {
  const isDarkMode = useDarkMode()
  const theme = isDarkMode ? 'dark' : 'light'

  const filtered = components.filter((c) => {
    return filterType != null && c.type === filterType
  })

  const getImgUrl = (fileName) => {
    return `/assets/components-thumbnails/${theme}/${fileName}`
  }


  return (
    <>
      <div className="sb-unstyled my-xl gap-lg flex-wrap flex">
        {filtered.length === 0 ? (
          <p className="text-body-1 text-neutral-outline">
            No components match &quot;{filter}&quot;. Try a different search.
          </p>
        ) : (
          filtered.map((component) => (
            <div key={component.name} className="flex flex-col gap-sm w-sz-184 shrink-0 min-h-[100px]">
              <a href={component.href} className="flex gap-md flex-col group">
                <TextLink underline={false} intent="support" className="text-headline-2">
                  {component.name}
                </TextLink>

                <div className="group-hover:scale-110 group-hover:shadow-sm transition-transform duration-150 w-[184px] h-[100px] bg-neutral-container rounded-lg flex items-center justify-center">
                  <img
                    src={getImgUrl(component.thumbnail)}
                    alt={`${component.name} component thumbnail`}
                    className="w-full h-full object-contain"
                    loading="lazy"
                    width="184"
                    height="100"
                  />
                </div>
              </a>
            </div>
          ))
        )}
      </div>
    </>

)
}

## Components

### Actions

Components that trigger user-initiated operations. They allow users to perform tasks such as submitting data, confirming choices, or initiating workflows. Typical examples include buttons, links, and menu items.

<FilterableComponentsList filterType={COMPONENTS_TYPES.ACTION} />


### Data inputs

Components that let users enter, edit, or select data. They are designed to capture user input in a structured way, such as text fields, checkboxes, radio buttons, selects, and date pickers.

<FilterableComponentsList filterType={COMPONENTS_TYPES.INPUT} />



### Indicators

Components that provide feedback about system status or ongoing processes. They help users understand what is happening, such as loading states, success/error messages, badges, or progress indicators.

<FilterableComponentsList filterType={COMPONENTS_TYPES.INDICATOR} />


### Layouts

Components that structure and organize content on the page. They define spatial relationships and alignment, helping create consistent and responsive page compositions (e.g. grids, stacks, containers, sections).

<FilterableComponentsList filterType={COMPONENTS_TYPES.LAYOUT} />


### Navigations

Components that help users move through the application and access different sections or views. They provide orientation and wayfinding, such as menus, tabs, breadcrumbs, and sidebars.

<FilterableComponentsList filterType={COMPONENTS_TYPES.NAVIGATION} />


### Overlays

Components that appear above the main content to capture attention or require interaction. They are typically used for temporary or contextual interactions, such as modals, dialogs, popovers, and tooltips.

<FilterableComponentsList filterType={COMPONENTS_TYPES.OVERLAY} />


### Visual assets

Components used to display non-interactive visual content that supports understanding or branding. This includes icons, illustrations, images, avatars, and logos.

<FilterableComponentsList filterType={COMPONENTS_TYPES.VISUAL_ASSET} />
